# merr
wrap golang error to make it easy to debug



# usage

```
package main

import (
	"errors"
	"fmt"

	"github.com/mozhata/merr"
)

func main() {
	var err error
	err = warpper()
	logMerr(err)
}

func warpper() error {
	err := errors.New("origin err")
	return merr.WrapErr(err, "new err")
}

func logMerr(err error) {
	e := merr.WrapErr(err)
	fmt.Printf("E%d: err: %s\nraw err: %s\ncall stack: %s\n",
		e.StatusCode,
		e.Error(),
		e.RawErr(),
		e.CallStack(),
	)
}

```

output:

```

raw err: origin err
```
